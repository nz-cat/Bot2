-- UTILITY FUNCTIONS
local function GetPlayers()
    local t = {}
    for _,p in ipairs(Players:GetPlayers()) do
        if p ~= lp then
            table.insert(t, p.Name)
        end
    end
    return t
end

local function GetNearestPlayer()
    local myChar = lp.Character
    if not myChar or not myChar:FindFirstChild("HumanoidRootPart") then return nil end

    local myHRP = myChar.HumanoidRootPart
    local nearest, shortest = nil, math.huge

    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= lp and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
            local hum = p.Character:FindFirstChildOfClass("Humanoid")
            if hum and hum.Health > 0 then
                local dist = (p.Character.HumanoidRootPart.Position - myHRP.Position).Magnitude
                if dist < shortest then
                    shortest = dist
                    nearest = p
                end
            end
        end
    end
    return nearest
end

local function getClosestCityNPC(hrp)
    local closest, dist = nil, math.huge
    for _, v in ipairs(Workspace:GetDescendants()) do

            local d = (v.HumanoidRootPart.Position - hrp.Position).Magnitude
            if d < dist then dist = d closest = v end
        end
    end
    return closest
end

local function bossTeleportToRandomLocation(npc)
    local char = lp.Character
    if not char then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    
    local currentTime = tick()
    if currentTime - bossLastTeleportTime >= bossTeleportDelay then
        local teleportLocations = getTeleportLocations()
        local randomIndex = math.random(1, #teleportLocations)
        bossCurrentTeleportIndex = randomIndex
        local location = teleportLocations[randomIndex]
        
        hrp.CFrame = CFrame.new(location)
        
        if npc and bossLookAtNPC2 then
            local root = npc:FindFirstChild("HumanoidRootPart")
            if root then
                local pos = hrp.Position
                local targetPos = Vector3.new(
                    root.Position.X,
                    pos.Y,
                    root.Position.Z
                )
                hrp.CFrame = CFrame.lookAt(pos, targetPos)
            end
        end
        
        bossLastTeleportTime = currentTime
        return randomIndex, location
    end
    return bossCurrentTeleportIndex, getTeleportLocations()[bossCurrentTeleportIndex]
end

-- AUTO LOOT
local lootedParts = {}
local LootCooldown = 1

task.spawn(function()
    while task.wait(0.1) do
        if not _G.AutoLootEnabled then
            table.clear(lootedParts)
            continue
        end
        
        local char = lp.Character
        if not char then continue end
        
        local hrp = char:FindFirstChild("HumanoidRootPart")
        if not hrp then continue end
        
        local overlapParams = OverlapParams.new()
        overlapParams.FilterType = Enum.RaycastFilterType.Blacklist
        overlapParams.FilterDescendantsInstances = {char}
        
        local lootWhitelist = {
            "coin","gold","money","cash","gem","diamond",
            "chest","treasure","ruong","rương",
            "drop","loot","item","box","crate"
        }
        
        local now = os.clock()
        
        local parts = Workspace:GetPartBoundsInRadius(
            hrp.Position,
            _G.LootDistance,
            overlapParams
        )
        
        for _, part in ipairs(parts) do
            if not part:IsA("BasePart") then continue end
            if not part.Parent then 
                lootedParts[part] = nil 
                continue 
            end
            
            if lootedParts[part] and now - lootedParts[part] < LootCooldown then
                continue
            end
            
            local name = part.Name:lower()
            local valid = false
            for _, k in ipairs(lootWhitelist) do
                if name:find(k, 1, true) then
                    valid = true
                    break
                end
            end
            if not valid then continue end
            
            local prompt = part:FindFirstChildWhichIsA("ProximityPrompt", true)
            if prompt and prompt.Enabled then
                fireproximityprompt(prompt)
                lootedParts[part] = now
                continue
            end
            
            if part:FindFirstChildWhichIsA("TouchTransmitter") then
                firetouchinterest(hrp, part, 0)
                firetouchinterest(hrp, part, 1)
                lootedParts[part] = now
                continue
            end
            
            local click = part:FindFirstChildWhichIsA("ClickDetector", true)
            if click then
                fireclickdetector(click)
                lootedParts[part] = now
            end
        end
    end
end)

-- TWEEN SYSTEM
local function StopTween()
    if TweenConnection then
        TweenConnection:Disconnect()
        TweenConnection = nil
    end
end

local function SmoothTweenTo(targetPos)
    StopTween()

    local character = lp.Character or lp.CharacterAdded:Wait()
    local hrp = character:WaitForChild("HumanoidRootPart")

    hrp.AssemblyLinearVelocity = Vector3.zero
    hrp.AssemblyAngularVelocity = Vector3.zero
    hrp.Anchored = true

    TweenConnection = RunService.Heartbeat:Connect(function(dt)
        if not hrp or not hrp.Parent then
            StopTween()
            return
        end

        local currentPos = hrp.Position
        local offset = targetPos - currentPos
        local distance = offset.Magnitude

        if distance <= STOP_DISTANCE then
            hrp.CFrame = CFrame.new(targetPos)
            StopTween()
            hrp.Anchored = false
            return
        end

        local direction = offset.Unit
        local moveStep = math.min(MOVE_SPEED * dt, distance)

        hrp.CFrame = hrp.CFrame + direction * moveStep
    end)
end

-- FIX LAG
local function ApplyLagFix(level)
    if level <= currentLagFixLevel then return end
    
    local Terrain = workspace:FindFirstChildOfClass("Terrain")
    
    if level >= 10 and currentLagFixLevel < 10 then
        Lighting.GlobalShadows = false
        Lighting.FogEnd = 9e9
    end
    
    if level >= 30 and currentLagFixLevel < 30 then
        for _, effect in ipairs(Lighting:GetChildren()) do
            if effect:IsA("BlurEffect") or effect:IsA("DepthOfFieldEffect") then
                effect.Enabled = false
            end
        end
    end
    
    if level >= 50 and currentLagFixLevel < 50 then
        for _, part in ipairs(workspace:GetDescendants()) do
            if part:IsA("BasePart") then
                part.Material = Enum.Material.SmoothPlastic
            end
        end
    end
    
    if level >= 70 and currentLagFixLevel < 70 then
        for _, obj in ipairs(workspace:GetDescendants()) do
            if obj:IsA("Decal") or obj:IsA("Texture") then
                obj:Destroy()
            elseif obj:IsA("MeshPart") then
                obj.TextureID = ""
            end
        end
    end
    
    currentLagFixLevel = level
end

-- BOSS FARM LOOP
task.spawn(function()
    while task.wait(0.1) do
        if not npc2Mode then
            task.wait(1)
            continue
        end
        
        local char = lp.Character
        if not char then continue end
        
        local hum = char:FindFirstChild("Humanoid")
        local hrp = char:FindFirstChild("HumanoidRootPart")
        if not hum or not hrp then continue end
        
        -- Auto Heal Boss
        if bossAutoHeal and hum.Health < hum.MaxHealth * 0.6 then
            local backpack = lp.Backpack
            if backpack then
                local bandage = backpack:FindFirstChild(BANDAGE_NAME)
                if bandage and bandage:IsA("Tool") then
                    local prevTool = char:FindFirstChildOfClass("Tool")
                    hum:EquipTool(bandage)
                    bandage:Activate()
                    task.wait(1.2)
                    if prevTool and prevTool.Parent == backpack then
                        hum:EquipTool(prevTool)
                    end
                end
            end
        end
        
        -- Auto Buy Bandage Boss
        if bossAutoBuyBandage and buyItemRemote then
            pcall(function()
                buyItemRemote:FireServer(BANDAGE_NAME, 99)
            end)
        end
        
        -- Find NPC2
        local npc = getNPC2()
        
        if npc then
            local npcHRP = npc:FindFirstChild("HumanoidRootPart")
            local npcHum = npc:FindFirstChild("Humanoid")
            
            if npcHRP and npcHum then
                -- Update UI
                bossHPLabel:Set(string.format("Boss HP: %d/%d", 
                    math.max(0, math.floor(npcHum.Health)), 
                    math.floor(npcHum.MaxHealth)))
                
                -- Create ESP
                if not bossESPBeam then
                    createBossESP(npcHRP)
                end
                
                -- Add glow to weapon
                local tool = char:FindFirstChildOfClass("Tool")
                if tool then
                    addBossGlow(tool)
                else
                    clearBossGlow()
                end
                
                -- Lock camera
                if bossLockCamera then
                    Camera.CameraType = Enum.CameraType.Custom
                    Camera.CameraSubject = npcHum
                else
                    Camera.CameraSubject = hum
                end
                
                -- Teleport and look at boss
                local locationIndex, location = bossTeleportToRandomLocation(npc)
                bossModeLabel:Set(string.format("Mode: %d | Vị trí: %d", bossFarmMode, locationIndex))
                
                bossStatusLabel:Set("Trạng thái: Đang đánh boss")
            end
        else
            removeBossESP()
            clearBossGlow()
            bossHPLabel:Set("Boss HP: Đang chờ respawn")
            bossStatusLabel:Set("Trạng thái: Chờ boss")
            
            -- Still teleport while waiting
            local locationIndex, location = bossTeleportToRandomLocation(nil)
            bossModeLabel:Set(string.format("Mode: %d | Vị trí: %d", bossFarmMode, locationIndex))
        end
        
        -- Update timer
        if bossFarmStartTime then
            local t = math.floor(tick() - bossFarmStartTime)
            bossTimerLabel:Set(string.format("Thời gian: %02d:%02d", math.floor(t / 60), t % 60))
        end
        
        -- Auto attack
        bossAutoAttack()
    end
end)

-- UI - PVP COMBAT TAB
local function toggleSpeedCFrame(v)
    if v then
        if speedVelConnection then return end
        speedVelConnection = RunService.Heartbeat:Connect(function(dt)
            if not SpeedBoost then return end
            local char = lp.Character
            if not char or not char.Parent then return end
            local hrp = char:FindFirstChild("HumanoidRootPart")
            local hum = char:FindFirstChildOfClass("Humanoid")
            if not hrp or not hum then return end
            if hum.MoveDirection.Magnitude > 0 then
                local oldY = hrp.AssemblyLinearVelocity.Y
                hrp.AssemblyLinearVelocity = Vector3.new(0, oldY, 0)
                hrp.CFrame = hrp.CFrame + (hum.MoveDirection * (WalkSpeed * dt))
                hrp:SetNetworkOwner(lp)
            end
        end)
    else
        if speedVelConnection then
            speedVelConnection:Disconnect()
            speedVelConnection = nil
        end
    end
end
