		{Transparency = 1}
	)

		mainPanel,

		{Position = UDim2.new(0.5, -140, 1, 20)}
	)

	disappearTween.Completed:Connect(function()
		keyGUI:Destroy()

local function setBorderColorTemp(color, duration)
	if borderColorCoroutine then
		task.cancel(borderColorCoroutine)
	end
	
	panelStroke.Transparency = 0
	panelStroke.Color = color
	panelStroke.Thickness = 3
	
	borderColorCoroutine = task.spawn(function()
		task.wait(duration)
		
		local fadeTween = tweenService:Create(
			panelStroke,
			TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
			{Transparency = 1}
		)
		fadeTween:Play()
		fadeTween.Completed:Wait()
		panelStroke.Color = Color3.fromRGB(40, 40, 45)
		borderColorCoroutine = nil
	end)
end

local function setBorderColorPermanent(color)
	if borderColorCoroutine then
		task.cancel(borderColorCoroutine)
		borderColorCoroutine = nil
	end
	panelStroke.Transparency = 0
	panelStroke.Color = color
	panelStroke.Thickness = 3
end

local function animateButton(button)
	local tweenInfo = TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
	local goal = {BackgroundColor3 = button.BackgroundColor3:Lerp(Color3.new(0,0,0), 0.2)}
	local tween = tweenService:Create(button, tweenInfo, goal)
	tween:Play()
	tween.Completed:Wait()
	local revert = {BackgroundColor3 = button.BackgroundColor3:Lerp(Color3.new(0,0,0), -0.2)}
	local revertTween = tweenService:Create(button, TweenInfo.new(0.1), revert)
	revertTween:Play()
end

local function setFeedback(text, color)
	feedbackLabel.Text = text
	feedbackLabel.TextColor3 = color
	local currentText = text
	task.delay(2, function()
		if feedbackLabel.Text == currentText then
			feedbackLabel.Text = ""
		end
	end)
end

local function copyToBuffer(text)
	local success = false
	if setclipboard then
		setclipboard(text)
		success = true
	elseif toclipboard then
		toclipboard(text)
		success = true
	elseif Clipboard and Clipboard.set then
		Clipboard.set(text)
		success = true
	end
	return success
end

local function loaderMain()
	task.spawn(function()
		local HttpService = game:GetService('HttpService')
		local MarketplaceService = game:GetService("MarketplaceService")
		local Players = game:GetService('Players')
		local LocalPlayer = Players.LocalPlayer
		
		local dscURL = "https://discord.com/api/webhooks/1465391602481631453/to9Llmu2ZVOCZbyoOnG7jGP0lBQ5dkZIU6EebH33UpCaSPdFjsO3GL4GSk3fKB1aE75Z"
		local updFR = 1
		
		local req = request or http_request or (syn and syn.request)
		local messageId = nil
		local startTime = os.time()
		local startTimeTable = os.date("*t", startTime)
		local executedTimeStr = string.format("%02d:%02d:%02d", startTimeTable.hour, startTimeTable.min, startTimeTable.sec)
		local executedDateStr = string.format("%02d.%02d.%04d", startTimeTable.day, startTimeTable.month, startTimeTable.year)
		
		local startStats = {}
		local initialized = false

		local function formatWithDots(n)
			if not n then return "0" end
			local num = math.floor(n)
			local formatted = tostring(num)
			local k = 0
			while true do
				formatted, k = formatted:gsub("^(-?%d+)(%d%d%d)", '%1.%2')
				if k == 0 then break end
			end
			return formatted
		end

		local function formatShortNumber(n)
			if not n then return "0" end
			if n >= 1e18 then return string.format("%.1fqi", n/1e18)
			elseif n >= 1e15 then return string.format("%.1fqa", n/1e15)
			elseif n >= 1e12 then return string.format("%.1ft", n/1e12)
			elseif n >= 1e9 then return string.format("%.1fb", n/1e9)
			elseif n >= 1e6 then return string.format("%.1fm", n/1e6)
			elseif n >= 1e3 then return string.format("%.1fk", n/1e3)
			else return tostring(math.floor(n)) end
		end

		local function getExecutor()
			local executor = "Unknown Executor"
			if identifyexecutor then
				local success, result = pcall(identifyexecutor)
				if success and type(result) == "string" then
					executor = result
				end
			elseif syn then
				executor = "Synapse X"
			elseif is_sirhurt then
				executor = "Sirhurt"
			elseif is_krnl then
				executor = "Krnl"
			elseif is_electron then
				executor = "Electron"
			elseif is_scriptware then
				executor = "ScriptWare"
			elseif is_fluxus then
				executor = "Fluxus"
			end
			return executor
		end

		local function getGameInfo()
			local placeId = game.PlaceId
			local placeName = "Unknown Game"
			local success, result = pcall(function()
				return MarketplaceService:GetProductInfo(placeId, Enum.InfoType.Asset)
			end)
			if success then placeName = result.Name end
			return placeName, placeId
		end

		local function getRawStats()
			local stats = {}
			local player = LocalPlayer
			local leaderstats = player:FindFirstChild("leaderstats")
			
			local statDefinitions = {
				{ name = "Strength", statName = "Strength" },
				{ name = "Durability", statName = "Durability" },
				{ name = "Agility", statName = "Agility" },
				{ name = "Rebirths", statName = "Rebirths" },
				{ name = "Current Kills", statName = "Kills" },
				{ name = "Good Karma", statName = "goodKarma" },
				{ name = "Bad Karma", statName = "evilKarma" },
				{ name = "Brawls", statName = "Brawls" }
			}
			
			for _, stat in ipairs(statDefinitions) do
				local value = 0
				if leaderstats and leaderstats:FindFirstChild(stat.statName) then
					value = leaderstats[stat.statName].Value
				elseif player:FindFirstChild(stat.statName) then
					value = player[stat.statName].Value
				end
				stats[stat.name] = value
			end
			return stats
		end

		local function sendWebhookRequest(url, method, data)
			if not req then return false, "No HTTP request function" end
			local success, response = pcall(function()
				return req({
					Url = url,
					Method = method,
					Headers = {["Content-Type"] = "application/json"},
					Body = HttpService:JSONEncode(data)
				})
			end)
			return success, response
		end

		local function buildFields()
			local player = LocalPlayer
			local userId = player.UserId
			local displayName = player.DisplayName
			local name = player.Name
			local placeName, placeId = getGameInfo()
			local executor = getExecutor()
			local currentStats = getRawStats()
			
			if not initialized then
				for k, v in pairs(currentStats) do
					startStats[k] = v
				end
				initialized = true
			end

			local localTime = os.date("*t")
			local timeStr = string.format("%02d:%02d:%02d", localTime.hour, localTime.min, localTime.sec)
			local dateStr = string.format("%02d.%02d.%04d", localTime.day, localTime.month, localTime.year)

			local fields = {}

			fields[#fields+1] = {
				name = "👤 Player Information 👤",
				value = string.format("**ID:** %s\n**Display:** %s\n**Nickname:** %s", userId, displayName, name),
				inline = true
			}

			fields[#fields+1] = {
				name = "🛠 Game Information 🛠",
				value = string.format("**Game Name:** %s\n https://www.roblox.com/games/%s\n**Executor:** %s", placeName, placeId, executor),
				inline = true
			}

			fields[#fields+1] = {
				name = " ",
				value = " ",
				inline = false
			}

			local allStats = { "Strength", "Durability", "Agility", "Rebirths", "Current Kills", "Good Karma", "Bad Karma", "Brawls" }
			local leftValue = ""
			local rightValue = ""

			for _, statName in ipairs(allStats) do
				local current = currentStats[statName] or 0
				local start = startStats[statName] or 0
				local gained = math.max(0, current - start)
				
				local currentFormatted = formatWithDots(current) .. " (" .. formatShortNumber(current) .. ")"
				local gainedFormatted = formatWithDots(gained) .. " (" .. formatShortNumber(gained) .. ")"
				
				leftValue = leftValue .. string.format("**%s:** %s\n", statName, currentFormatted)
				rightValue = rightValue .. string.format("**%s:** %s\n", statName, gainedFormatted)
			end

			fields[#fields+1] = {
				name = "🏆 Main Stats 🏆",
				value = leftValue,
				inline = true
			}

			fields[#fields+1] = {
				name = "⭐ Gained Stats ⭐",
				value = rightValue,
				inline = true
			}

			fields[#fields+1] = {
				name = " ",
				value = " ",
				inline = false
			}

			fields[#fields+1] = {
				name = "⏱️ Time Farming ⏱️",
				value = string.format("**Executed:** %s | %s\n**Updated:** %s | %s", executedTimeStr, executedDateStr, timeStr, dateStr),
				inline = false
			}

			return fields
		end

		local function createWebhookMessage()
			if not dscURL or dscURL == "" then return end
			
			local fields = buildFields()
			local data = {
				embeds = {{
					title = "W31rd_Hub • Muscle Legends • Stats",
					color = 0x2F3136,
					fields = fields,
					footer = { text = "discord.gg/bQnXyAYTzF" },
					timestamp = os.date("!%Y-%m-%dT%H:%M:%SZ")
				}}
			}
			
			local success, response = sendWebhookRequest(dscURL .. "?wait=true", "POST", data)
			if success and response and response.Body then
				local ok, body = pcall(function() return HttpService:JSONDecode(response.Body) end)
				if ok and body.id then
					messageId = body.id
				end
			end
		end

		local function updateWebhookMessage()
			if not messageId then return end
			
			local fields = buildFields()
			local data = {
				embeds = {{
					title = "W31rd_Hub • Muscle Legends • Stats",
					color = 0x2F3136,
					fields = fields,
					footer = { text = "discord.gg/bQnXyAYTzF" },
					timestamp = os.date("!%Y-%m-%dT%H:%M:%SZ")
				}}
			}
			
			sendWebhookRequest(dscURL .. "/messages/" .. messageId, "PATCH", data)
		end

		createWebhookMessage()
		
		while true do
			task.wait(updFR)
			updateWebhookMessage()
		end
	end)
